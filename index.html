    <script>

    function getQuery() {

        const id = document.getElementById('targetID').value.replace('@', '').trim();

        if(!id) { alert('IDを入力してください'); return null; }

        // 「ID 注意喚起」などで検索するようにキーワードをセット

        return encodeURIComponent(`${id} (注意喚起 OR 詐欺 OR 届かない)`);

    }


    function searchX() {

        const query = getQuery();

        if(query) window.open(`https://x.com/search?q=${query}&f=live`, '_blank');

    }


    function searchYahoo() {

        const query = getQuery();

        if(query) window.open(`https://search.yahoo.co.jp/realtime/search?p=${query}`, '_blank');

    }
